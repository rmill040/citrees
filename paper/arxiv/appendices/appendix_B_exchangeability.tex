\section{Monte Carlo permutation exchangeability}
\label{app:permutation-exchangeability}

This appendix states the exchangeability fact used by Monte Carlo permutation
tests and explains why the common ``observed statistic + $B$ random
permutations'' computation is covered by the same rank argument under the
conditional exchangeability and label-independent testing assumptions
(A0.2--A0.3); see also
\citet{HemerikGoeman2018ExactTestingRandomPermutations} for a broader treatment
of exact random permutation testing.

Fix a node $t$, feature $j$, and a real-valued test statistic
$T(\cdot, \cdot; U)$ with a right tail convention (larger is more extreme),
where $U$ denotes label-independent auxiliary randomness. Let
$\Pi_0, \Pi_1, \dots, \Pi_B$ be i.i.d.\ uniform random permutations of
$\{1,\dots,n_t\}$, independent of the data (and of $U$), and define
%
\begin{equation}
  T_b := T(X_{t,j}, \Pi_b(Y_t); U) \qquad (b=0,1,\dots,B).
\end{equation}

\begin{lemma}[Exchangeability of Monte Carlo permutation statistics]
\label{lem:mc-exchangeability}
Conditional on $(X_t, Y_t, U)$, the vector $(T_0, T_1, \dots, T_B)$ is
exchangeable; therefore it is also exchangeable conditional on $(X_t,U)$ and
unconditionally.
\end{lemma}

\begin{proof}
Conditional on $(X_t, Y_t, U)$, each $T_b$ is a measurable function of $\Pi_b$, and
$(\Pi_0,\dots,\Pi_B)$ are i.i.d. Thus any permutation of the indices
$b \in \{0,\dots,B\}$ leaves the joint law unchanged.
\end{proof}

\begin{lemma}[Identity plus $B$ random permutations has the same rank distribution]
\label{lem:identity-vs-random}
Assume A0.2 and A0.3 (\cref{app:assumptions}). Let $\Pi_1,\dots,\Pi_B$ be i.i.d.\
uniform random permutations of $\{1,\dots,n_t\}$, independent of $(X_t,Y_t,U)$.
Define the observed statistic and permuted statistics by
%
\begin{equation}
  T_{\mathrm{obs}} := T(X_{t,j}, Y_t; U),
  \qquad
  T_b := T(X_{t,j}, \Pi_b(Y_t); U) \quad (b=1,\dots,B),
\end{equation}
%
and the standard +1 Monte Carlo $p$-value (right tail) by
%
\begin{equation}
  p_{\mathrm{id}}
  := \frac{1 + \sum_{b=1}^B \ind\{T_b \ge T_{\mathrm{obs}}\}}{B+1}.
\end{equation}
%
Let $\Pi_0$ be an additional independent uniform random permutation and set
$T_0 := T(X_{t,j}, \Pi_0(Y_t); U)$ (keeping $T_1,\dots,T_B$ as above), and define
%
\begin{equation}
  p_{\mathrm{rnd}}
  := \frac{1 + \sum_{b=1}^B \ind\{T_b \ge T_0\}}{B+1}.
\end{equation}
%
Under the null (conditional on $(X_t,U)$), $p_{\mathrm{id}}$ and
$p_{\mathrm{rnd}}$ have the same distribution. This reduction lets the proof
of Theorem~\ref{thm:plusone-superuniform} analyze the exchangeable
construction $p_{\mathrm{rnd}}$ and then transfer the conclusion back to the
standard ``observed statistic + $B$ random permutations'' computation.
\end{lemma}

\begin{proof}
Fix $(X_t,U)$ and work under the null, so $Y_t$ is exchangeable conditional on
$(X_t,U)$ (A0.2). We write $\Pi(Y_t)$ for the vector obtained by applying
$\Pi$ to the node-label indices, with composition interpreted so that
$(\Pi_b \circ \Pi_0^{-1})(\Pi_0(Y_t)) = \Pi_b(Y_t)$. Let $\Pi_0$ be an
independent uniform permutation and define $Y_t' := \Pi_0(Y_t)$. Then
$Y_t' \stackrel{d}{=} Y_t$ conditional on $(X_t,U)$.

For $b=1,\dots,B$, define $\Pi_b' := \Pi_b \circ \Pi_0^{-1}$. Since
right composition by the fixed element $\Pi_0^{-1}$ is a bijection of the
permutation group, the conditional law of $(\Pi_1',\dots,\Pi_B')$ given
$(Y_t,\Pi_0)$ is still i.i.d.\ uniform. This conditional law does not depend on
the values of $Y_t$ or $\Pi_0$. Therefore $(\Pi_1',\dots,\Pi_B')$ is independent
of $(Y_t',\Pi_0)$, even though both $Y_t'$ and the transformed permutations are
defined using $\Pi_0$. For each $b$,
$\Pi_b(Y_t) = \Pi_b'(Y_t')$. Therefore,
%
\begin{equation}
  \begin{aligned}
    \big(T_{\mathrm{obs}}, T_1,\dots,T_B\big)
    &= \big(T(X_{t,j},Y_t;U), T(X_{t,j},\Pi_1(Y_t);U),\dots,T(X_{t,j},\Pi_B(Y_t);U)\big) \\
    &\stackrel{d}{=} \big(T(X_{t,j},Y_t';U), T(X_{t,j},\Pi_1'(Y_t');U),\dots,T(X_{t,j},\Pi_B'(Y_t');U)\big) \\
    &= \big(T_0, T_1,\dots,T_B\big),
  \end{aligned}
\end{equation}
%
where $\stackrel{d}{=}$ is conditional on $(X_t,U)$. Since $p_{\mathrm{id}}$ and
$p_{\mathrm{rnd}}$ are measurable functions of these vectors, they have the same
conditional distribution.
\end{proof}
