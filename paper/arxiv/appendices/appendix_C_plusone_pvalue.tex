\section{Proof of Theorem~\ref{them:plusone-superuniform}}
\label{app:proof-plusone}

\paragraph{Claim.} Theorem~\ref{them:plusone-superuniform}.
\paragraph{Assumptions used.} A0.1--A0.5 (\cref{app:assumptions}).

Fix $(X_t,U)$ and work under the null hypothesis at the fixed node $t$. Under
A0.2--A0.3 (\cref{app:assumptions}), the standard computation
``$T_{\mathrm{obs}}$ + $B$ random permutations'' is covered by the same rank
argument as the i.i.d.\ permutation construction; see
\cref{lem:identity-vs-random} in \cref{app:permutation-exchangeability} and
\citet{HemerikGoeman2018ExactTestingRandomPermutations}. In this
proof, it therefore suffices to analyze the exchangeable construction in which
the vector $(T_0,T_1,\dots,T_B)$ is exchangeable conditional on $(X_t,U)$.

Define the (right-tail) upper rank
%
\begin{equation}
  R := 1 + \sum_{b=1}^B \ind\{T_b \ge T_0\},
  \qquad
  p_{\mathrm{MC}}^{\ge} = \frac{R}{B+1}.
\end{equation}
%
The left-tail variant (used in Stage~B) replaces the exceedance check by
$\ind\{T_b \le T_0\}$; equivalently, apply the right-tail argument below to
$-T$.

\paragraph{Tie-free case.}
If the statistics are almost surely tie-free, then $R = 1 + \sum_{b=1}^B \ind\{T_b > T_0\}$
is the (descending) rank of $T_0$ among $(T_0,\dots,T_B)$. Exchangeability implies
%
\begin{equation}
  \PP(R = r \mid X_t,U) = \frac{1}{B+1}, \qquad r=1,\dots,B+1,
\end{equation}
%
so for any $\alpha \in [0,1]$,
%
\begin{equation}
  \PP(p_{\mathrm{MC}}^{\ge} \le \alpha \mid X_t,U)
  = \PP\!\big(R \le (B+1)\alpha \mid X_t,U\big)
  = \frac{\lfloor (B+1)\alpha \rfloor}{B+1}
  \le \alpha.
\end{equation}

\paragraph{Ties.}
If ties can occur, we use the tie convention A0.5. Under conservative tie
handling, p-values are computed with the indicator $\ind\{T_b \ge T_0\}$, which
counts ties as exceedances. Let $V_0,\dots,V_B \stackrel{\mathrm{i.i.d.}}{\sim}
\mathrm{Unif}(0,1)$ be independent of all other randomness and define augmented
statistics $\tilde{T}_b := (T_b, V_b)$ ordered lexicographically. Then
$(\tilde{T}_0,\dots,\tilde{T}_B)$ is exchangeable and almost surely tie-free, so
the rank argument above gives a super-uniform p-value
%
\begin{equation}
  \tilde{p}_{\mathrm{MC}}^{\ge} := \frac{1 + \sum_{b=1}^B \ind\{\tilde{T}_b \ge_{\mathrm{lex}} \tilde{T}_0\}}{B+1}.
\end{equation}
%
Moreover, for each $b$, $\ind\{\tilde{T}_b \ge_{\mathrm{lex}} \tilde{T}_0\} \le \ind\{T_b \ge T_0\}$,
so $p_{\mathrm{MC}}^{\ge} \ge \tilde{p}_{\mathrm{MC}}^{\ge}$ and hence
%
\begin{equation}
  \PP(p_{\mathrm{MC}}^{\ge} \le \alpha \mid X_t,U)
  \le \PP(\tilde{p}_{\mathrm{MC}}^{\ge} \le \alpha \mid X_t,U) \le \alpha.
\end{equation}
%
Taking expectations over $(X_t,U)$ yields $\PP(p_{\mathrm{MC}}^{\ge} \le \alpha) \le \alpha$.
