\section{Proof of Proposition~\ref{prop:per-feature-bound}}
\label{app:proof-per-feature}

\paragraph{Claim.} Proposition~\ref{prop:per-feature-bound}.
\paragraph{Assumptions used.} A0.1--A0.5 (\cref{app:assumptions}).

If node $t$ accepts a split on feature $j$, then Stage~A must have rejected for
$j$ at the strict Bonferroni threshold, hence
$p_{t,j} < \alpha_{\mathrm{sel}}/m_t$. Therefore,
%
\begin{equation}
  \begin{aligned}
    \PP(\text{node $t$ accepts a split on $j$} \mid X_t,U)
    &\le \PP(p_{t,j} < \alpha_{\mathrm{sel}}/m_t \mid X_t,U) \\
    &\le \PP(p_{t,j} \le \alpha_{\mathrm{sel}}/m_t \mid X_t,U) \\
    &\le \alpha_{\mathrm{sel}}/m_t,
  \end{aligned}
\end{equation}
%
by super-uniformity of $p_{t,j}$ under the complete null. This proves the
conditional claim in Proposition~\ref{prop:per-feature-bound}. We do not collapse
the bound to the same scalar form after averaging over $(X_t,U)$, because
$m_t$ is itself $(X_t,U)$-measurable under A0.3.

\subsection{Proof of Corollary~\ref{cor:subsampling-per-feature}}
\label{app:proof-subsampling-per-feature}

\paragraph{Claim.} Corollary~\ref{cor:subsampling-per-feature}.
\paragraph{Assumptions used.} A0.1--A0.5 (\cref{app:assumptions}) and uniform
without-replacement feature subsampling from a full candidate set of size $\bar m_t$.

Condition on $X_t$ and treat the full candidate set $\bar F_t$ and its size
$\bar m_t$ as fixed. Let $F_t$ be the tested subset obtained by sampling $m_t$
features uniformly without replacement from $\bar F_t$, independently of $Y_t$.

If $j \notin F_t$ then node $t$ cannot accept a split on $j$. If $j \in F_t$, then
Proposition~\ref{prop:per-feature-bound} applied to the tested family $F_t$
implies
%
\begin{equation}
  \PP(\text{node $t$ accepts a split on $j$} \mid X_t,F_t) \le \alpha_{\mathrm{sel}}/m_t.
\end{equation}
%
Taking expectations over $F_t$ conditional on $X_t$ yields
%
\begin{equation}
  \begin{aligned}
    \PP(\text{node $t$ accepts a split on $j$} \mid X_t)
    &= \E\!\left[\PP(\text{node $t$ accepts a split on $j$} \mid X_t,F_t)\,\middle|\, X_t\right] \\
    &\le \PP(j \in F_t \mid X_t)\cdot \alpha_{\mathrm{sel}}/m_t
    = (m_t/\bar m_t)\cdot \alpha_{\mathrm{sel}}/m_t
    = \alpha_{\mathrm{sel}}/\bar m_t,
  \end{aligned}
\end{equation}
%
since under uniform subsampling $\PP(j \in F_t \mid X_t) = m_t/\bar m_t$.
