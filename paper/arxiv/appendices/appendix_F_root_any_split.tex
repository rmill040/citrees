\section{Proof of Proposition~\ref{prop:any-split-implies-root-rejection}}
\label{app:proof-root}

\paragraph{Claim.} Proposition~\ref{prop:any-split-implies-root-rejection}.
\paragraph{Assumptions used.} A0.1--A0.5 at the root (\cref{app:assumptions}).

The tree can only contain any internal split if the root passes Stage~A, so the
event ``tree has at least one internal split'' is a subset of ``root Stage~A
rejects.'' Under a complete global null at the root, \cref{prop:stageA-global-null}
applies at $t = \mathrm{root}$, yielding the stated bound.

\subsection{Proof of Corollary~\ref{cor:forest-level-root}}
\label{app:proof-forest-level-root}

\paragraph{Claim.} Corollary~\ref{cor:forest-level-root}.
\paragraph{Assumptions used.} A0.1--A0.5 at the root for each tree, a complete
global null over tested root features (per tree), and (for the final bound)
uniform feature subsampling as in \cref{cor:subsampling-per-feature}.

Let $S_m := \ind\{\text{tree $m$ has at least one internal split}\}$ and
$N_{\mathrm{split}} := \sum_{m=1}^M S_m$.
By Proposition~\ref{prop:any-split-implies-root-rejection}, for each tree $m$,
the event $\{S_m=1\}$ implies ``root Stage~A rejects'' for that tree. Under the
complete root null, Proposition~\ref{prop:stageA-global-null} gives
$\PP(S_m=1) \le \alpha_{\mathrm{sel}}$. Therefore
%
\begin{equation}
  \E[N_{\mathrm{split}}]
  = \sum_{m=1}^M \E[S_m]
  = \sum_{m=1}^M \PP(S_m=1)
  \le M\,\alpha_{\mathrm{sel}}.
\end{equation}
%
Moreover, by the union bound,
%
\begin{equation}
  \PP(N_{\mathrm{split}} \ge 1)
  = \PP\!\left(\bigcup_{m=1}^M \{S_m=1\}\right)
  \le \sum_{m=1}^M \PP(S_m=1)
  \le M\,\alpha_{\mathrm{sel}},
\end{equation}
%
and the upper bound can be truncated at $1$.

For the per-feature statement, let
$R_{m,j} := \ind\{\text{tree $m$ splits at the root on feature $j$}\}$. Under
uniform root feature subsampling from $p$ available features and the complete
root null, Corollary~\ref{cor:subsampling-per-feature} gives
$\PP(R_{m,j}=1) \le \alpha_{\mathrm{sel}}/p$ for each tree $m$. Hence
%
\begin{equation}
  \E\!\left[\sum_{m=1}^M R_{m,j}\right]
  = \sum_{m=1}^M \PP(R_{m,j}=1)
  \le M\,\alpha_{\mathrm{sel}}/p,
\end{equation}
%
again without requiring any independence across trees.
