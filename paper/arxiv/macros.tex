% Common macros and theorem environments for the arXiv manuscript.

% Paper requirement: avoid naming implementation libraries in the manuscript.
% Use \citrees as a neutral stand-in for the implementation used in experiments.
\newcommand{\citrees}{the reference implementation}

% \Pr is defined by amsmath (as an operator), so use \PP for probability.
\newcommand{\PP}{\mathbb{P}}
\newcommand{\ind}{\mathbf{1}}

\DeclareMathOperator*{\argmin}{arg\,min}
\DeclareMathOperator*{\uargmin}{uarg\,min}

\theoremstyle{plain}
\newtheorem{theorem}{Theorem}
\newtheorem{lemma}{Lemma}
\newtheorem{corollary}{Corollary}

\theoremstyle{definition}
\newtheorem{definition}{Definition}

\theoremstyle{remark}
\newtheorem{remark}{Remark}
