\section{Theoretical guarantees}
\label{sec:theory}

The theorem-backed object in this paper is the exhaustive fixed-$B$ Stage~A
feature-selection rule at a pre-specified node. All probability statements below are
conditional on $(X_t,U)$.

\begin{theorem}[$+1$ permutation p-values are super-uniform]
\label{thm:plusone-superuniform}
Assume A0.1--A0.5 as listed in the supplement, so that under the tested null the
permutation statistics $(T_0,T_1,\dots,T_B)$ are exchangeable conditional on
$(X_t,U)$. Define the right-tail $+1$ Monte Carlo p-value
%
\begin{equation}
  p_{\mathrm{MC}}^{\ge}
  := \frac{1 + \sum_{b=1}^B \ind\{T_b \ge T_0\}}{B+1}.
\end{equation}
%
Then for every $\alpha \in [0,1]$,
%
\begin{equation}
  \PP(p_{\mathrm{MC}}^{\ge} \le \alpha \mid X_t,U) \le \alpha.
\end{equation}
\end{theorem}

This is the standard exchangeable-rank result for Monte Carlo permutation
tests. The supplement gives the proof. The left-tail version follows by
negating $T$.

\begin{corollary}[Stage~A complete-null control at a fixed node]
\label{cor:stageA-global-null}
Assume A0.1--A0.5 as listed in the supplement. Under the nodewise complete null
at $t$, let Stage~A reject when
%
\begin{equation}
  \min_{j \in F_t} p_{t,j} < \alpha_{\mathrm{sel}}/m_t.
\end{equation}
%
Then
%
\begin{equation}
  \PP\!\left(\exists j \in F_t : p_{t,j} < \alpha_{\mathrm{sel}}/m_t \,\middle|\, X_t,U\right)
  \le \alpha_{\mathrm{sel}}.
\end{equation}
\end{corollary}

This is an immediate union-bound consequence of
\cref{thm:plusone-superuniform}. It controls fixed-node Stage~A feature
selection under
the complete null; partial-null FWER, post-selection calibration of
$j_t^\star$, and end-to-end guarantees for the full adaptive learner are
separate targets.

\begin{remark}[Root-level structural consequence]
\label{rem:root-null}
If a tree grown in fixed-$B$ mode contains at least one internal split, then the
root must have rejected in Stage~A. Under the complete root null,
\cref{cor:stageA-global-null} therefore gives
%
\begin{equation}
  \PP(\text{the fitted tree has at least one internal split} \mid X_{\mathrm{root}},U)
  \le \alpha_{\mathrm{sel}}.
\end{equation}
\end{remark}
