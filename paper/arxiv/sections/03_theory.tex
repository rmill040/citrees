\section{Fixed-node Stage~A guarantees}
\label{sec:theory}

The theorem studies the exhaustive fixed-$B$ Stage~A
feature-selection rule at a fixed node. All probabilities below are
conditional on $(X_t,U)$, where $U$ denotes any auxiliary randomness held fixed
when conditioning.

\begin{theorem}[$+1$ permutation p-values are super-uniform]
\label{thm:plusone-superuniform}
Assume A0.1--A0.5 from \cref{app:assumptions}, so that under the tested null the
permutation statistics $(T_0,T_1,\dots,T_B)$ are exchangeable conditional on
$(X_t,U)$. Define the right-tail $+1$ Monte Carlo p-value
%
\begin{equation}
  p_{\mathrm{MC}}^{\ge}
  := \frac{1 + \sum_{b=1}^B \ind\{T_b \ge T_0\}}{B+1}.
\end{equation}
%
Then for every $\alpha \in [0,1]$,
%
\begin{equation}
  \PP(p_{\mathrm{MC}}^{\ge} \le \alpha \mid X_t,U) \le \alpha.
\end{equation}
\end{theorem}

This is the standard exchangeable-rank result for Monte Carlo permutation
tests \citep{NorthCurtisSham2002EmpiricalPValuesMonteCarlo,Ernst2004PermutationMethodsExactInference,HemerikGoeman2018ExactTestingRandomPermutations}; \cref{app:permutation-exchangeability,app:proof-plusone} give the
exchangeability statement and proof. The left-tail version follows by
applying the same result to $-T$.

\begin{corollary}[Stage~A complete-null control at a fixed node]
\label{cor:stageA-global-null}
Assume A0.1--A0.5 from \cref{app:assumptions}. Under the nodewise complete null
at $t$, let Stage~A reject when
%
\begin{equation}
  \min_{j \in F_t} p_{t,j} < \alpha_{\mathrm{sel}}/m_t.
\end{equation}
%
Then
%
\begin{equation}
  \PP\!\left(\exists j \in F_t : p_{t,j} < \alpha_{\mathrm{sel}}/m_t \,\middle|\, X_t,U\right)
  \le \alpha_{\mathrm{sel}}.
\end{equation}
\end{corollary}

This follows immediately from
\cref{thm:plusone-superuniform}. It controls fixed-node Stage~A feature
selection under the complete null. It does not cover partial-null FWER,
post-selection calibration of $j_t^\star$, or end-to-end guarantees for the
full adaptive learner.

\begin{remark}[Root-level implication]
\label{rem:root-null}
If a fixed-$B$ tree has at least one internal split, then the root must have
passed the Stage~A threshold. Under the complete root null,
\cref{cor:stageA-global-null} gives
%
\begin{equation}
  \PP(\text{the fitted tree has at least one internal split} \mid X_{\mathrm{root}},U)
  \le \alpha_{\mathrm{sel}}.
\end{equation}
\end{remark}
